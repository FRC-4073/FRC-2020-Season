public class test {

  /*
  * DESCRIPTION
  *
  * @author <author(add more if necessary)>
  * 
  * @date_created <DATE>
  * @date_modified <DATE>
  *
  * @revision <REVISION #>
  *
  * OTHER METHOD TAGS
  */

  public static void main(String[] args) {
    
    // Hold value describing my boomer status
    boolean iamboomer = false;
    
    // Outputs a message depending on my boomer status
    if (iamboomer) {
    
      // TRUE, I am a boomer
      System.out.println("I am a boomer");
      
    } else {
    
      // ELSE / FALSE, I am not a boomer
      System.out.println("I am not a boomer");
     
    }
  }
}
